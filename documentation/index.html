Customize Slider

If you want to change the slider's speed transition, open up the file argo-slide.js and change the following code: 

jQuery(function($){
    /**
     * Init slide
     * --------
     */
    var options = {
        nextButton: true,
        prevButton: true,
       // pagination: true,
        animateStartingFrameIn: true,
      //  autoPlay: true,
        autoPlayDelay: 3000,
    };
    var mySequence = $("#sequence").sequence(options).data("sequence");
})
